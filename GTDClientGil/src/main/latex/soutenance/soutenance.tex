	\subsection{Gestion de la persistance}
	\frame
	{
		\frametitle{Le choix Hibernate}

		Notre choix :
		\begin{itemize}
			\item Utilisation d'une base de données embarquée
			\item Mapping géré par des fichiers {\rmfamily\textsc{xml}}
		\end{itemize}

		\pause

		En conséquence :
		\begin{itemize}
			\item Pas de déploiement sur un serveur (JBoss par exemple)
			\item Pas besoin d'utiliser {\rmfamily\textsc{jdbc}}
		\end{itemize}
	}
	
	\frame
	{
		\frametitle{La base de données H2}
		\begin{itemize}
			\item Très rapide, open source, possiblité d'utiliser l'API {\rmfamily\textsc{jdbc}}
			\item Utilisation en mode embarqué ou en mode serveur
			\item Console basée sur un navigateur
			\item Langage de requête ressemblant à Oracle
			\item Notre application ne nécessite pas de stocker énormement de données 
		\end{itemize}
	}

	\begin{frame}[fragile]
		\frametitle{Exemple d'utilisation du patron DAO}
		\begin{exampleblock}{Récupérer les tâches d'un projet spécifique}
			\begin{lstlisting}[basicstyle=\small]
@Override
public List<Tache> recupererTaches(Long id) {
  Session s = getSession();
  Query q = s.createQuery("from Tache where idprojet = :idprojet");
  q.setParameter("idprojet", id);
  List<Tache> results = (List<Tache>) q.list();
  return results;
}
